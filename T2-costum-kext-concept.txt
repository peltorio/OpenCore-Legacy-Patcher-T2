2. Die Struktur der "Custom-Erweiterung" (Kext)
Ein solcher Patch wird meist als Plugin für Lilu geschrieben. Hier ist der konzeptionelle Aufbau (Pseudo-C++), den Entwickler für Tahoe nutzen:

C++
#include <Headers/kern_api.hpp>
#include <Headers/kern_patcher.hpp>

// Das Ziel-Symbol im AppleT2Controller, das die Version prüft
static const char *symbolBridgeVersion = "__ZN17AppleT2Controller16getBridgeVersionEv";

// Unsere Ersatz-Funktion
uint64_t myHookedBridgeVersion(void *instance) {
    // Wir täuschen die Version vor, die macOS 26 Tahoe verlangt
    // Entspricht bridgeOS 23.16.x in Hex-Form
    return 0x1710334000000000; 
}

void patchT2(void *user, KernelPatcher &patcher) {
    auto kext = patcher.loadKext("com.apple.driver.AppleT2Controller");
    if (kext) {
        // Lilu leitet den Funktionsaufruf auf unsere Lüge um
        patcher.routeFunction(symbolBridgeVersion, myHookedBridgeVersion, true);
    }
}
